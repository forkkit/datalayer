This demonstrates `React.js` hooks working for a Button.

export const WithCounter = () => {
  const [counter, setCounter] = useState(0);
  const label = `Counter: ${counter}`;
  return <Button variant="outlined" color="secondary" onClick={() => setCounter(counter + 1)}>{label}</Button>;
};
